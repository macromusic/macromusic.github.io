 '定型Ｐ_指定Windowの画面サイズを設定する
Private Sub PfixSetWindowPositionSize(myXbisExitFlag As Boolean, _
            myXobjWindw As Object, _
            Optional coXdouTopPrcnt As Double = 0, _
            Optional coXdouLftPrcnt As Double = 0, _
            Optional coXdouHghtPrcnt As Double = 100, _
            Optional coXdouWdthPrcnt As Double = 100)
'coXdouTopPrcnt : 0～100[%]の数値で入力
'coXdouLftPrcnt : 0～100[%]の数値で入力
'coXdouHghtPrcnt : 0～100[%]の数値で入力
'coXdouWdthPrcnt : 0～100[%]の数値で入力
    myXbisExitFlag = False
    If myXobjWindw Is Nothing Then Set myXobjWindw = ActiveWindow
    If coXdouHghtPrcnt < 0 Or coXdouHghtPrcnt > 100 Then GoTo ExitPath
    If coXdouWdthPrcnt < 0 Or coXdouWdthPrcnt > 100 Then GoTo ExitPath
  Dim myXdouMaxWdth As Double, myXdouMaxHght As Double
    With Application
        .WindowState = xlMaximized
        myXdouMaxHght = .Height: myXdouMaxWdth = .Width
    End With
    With myXobjWindw
        .WindowState = xlNormal
        .Top = coXdouTopPrcnt / 100 * myXdouMaxHght
        .Left = coXdouLftPrcnt / 100 * myXdouMaxWdth
        .Height = coXdouHghtPrcnt / 100 * myXdouMaxHght
        .Width = coXdouWdthPrcnt / 100 * myXdouMaxWdth
    End With
ExitPath:
    myXbisExitFlag = True
End Sub
