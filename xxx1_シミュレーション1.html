'SnsP_シート上データを取得
Private Sub snsProc()
    myXbisExitFlag = False
    
'//シート上のデータとコメントから文字列を検索してデータと位置情報を取得
    Call MsubSrchShtCmnt.callProc(myXlonTrgtValCnt, myZstrTrgtVal, myZobjTrgtRng)
    If myXlonTrgtValCnt <= 0 Then Exit Sub
    
'//シート上の指定範囲またはデータ全範囲から数値データ配列を取得
  Dim Li As Long: Li = LBound(myZobjTrgtRng, 1)
  Dim i As Long, j As Long
    i = myZobjTrgtRng(Li).Row
    j = myZobjTrgtRng(Li).Column + 1
    With myXobjSheet
        Set myXobjFrstCell = .Cells(i, j)
        Set myXobjLastCell = .Cells(.Rows.Count, j).End(xlUp)
    End With
    
  Dim myXbisCompFlag As Boolean
    Call MsubShtNumAry.callProc( _
            myXbisCompFlag, _
            myXlonNumRowCnt1, myXlonNumColCnt1, myZdouNumData1, _
            myXstrHdrRowCnt1, myXstrHdrColCnt1, myZstrHdr1, _
            myXobjSheet, myXobjFrstCell, myXobjLastCell)
    If myXbisCompFlag = False Then GoTo ExitPath
    
'//シート上の指定範囲またはデータ全範囲から数値データ配列を取得
    i = myZobjTrgtRng(Li).Row
    j = myZobjTrgtRng(Li).Column + 2
    With myXobjSheet
        Set myXobjFrstCell = .Cells(i, j)
        Set myXobjLastCell = .Cells(.Rows.Count, j).End(xlUp)
    End With
    
    Call MsubShtNumAry.callProc( _
            myXbisCompFlag, _
            myXlonNumRowCnt2, myXlonNumColCnt2, myZdouNumData2, _
            myXstrHdrRowCnt2, myXstrHdrColCnt2, myZstrHdr2, _
            myXobjSheet, myXobjFrstCell, myXobjLastCell)
    If myXbisCompFlag = False Then GoTo ExitPath
    
    Exit Sub
ExitPath:
    myXbisExitFlag = True
End Sub

'PrcsP_マクロを使用して信号処理を実行
Private Sub prsProc()
    myXbisExitFlag = False
    
  Dim i As Long
    i = UBound(myZstrTrgtVal): ReDim myZvarCalPara(i) As Variant
    For i = LBound(myZstrTrgtVal) To UBound(myZstrTrgtVal)
        myZvarCalPara(i) = myZstrTrgtVal(i)
    Next i
    
  Dim myXbisCompFlag As Boolean
    Call m2Msub1SgnlPrcs.callProc( _
            myXbisCompFlag, myXlonCalDataCnt1, myZvarCalData1, myXlonHdrCnt1, myZstrHeader1, _
            myZdouNumData1, myZvarCalPara)
    If myXbisCompFlag = False Then GoTo ExitPath
    
    Call m2Msub2SgnlPrcs.callProc( _
            myXbisCompFlag, _
            myXlonCalDataCnt2, myZvarCalData2, myXlonHdrCnt2, myZstrHeader2, _
            myXlonCalDataCnt3, myZvarCalData3, myXlonHdrCnt3, myZstrHeader3, _
            myXlonCalDataCnt4, myZvarCalData4, myXlonHdrCnt4, myZstrHeader4, _
            myZdouNumData1, myZdouNumData2, myZvarCalData1, myZvarCalPara)
    If myXbisCompFlag = False Then GoTo ExitPath
    
    Exit Sub
ExitPath:
    myXbisExitFlag = True
End Sub
