'Includes bMsub1MultiToOne
'Includes bMsub2OneToMulti
'Includes bMsub3MultiToMulti

Option Explicit
Option Base 1

'◆ModuleProc名_ファイルからファイルへデータをコピペする
'Rev.001
  
'//モジュールメモリ
  Private Const meMstrMdlName As String = "bMexeFileToFile"
  
'//出力制御信号
  Private myXbisCmpltFlag As Boolean
  
'//モジュール内変数_制御信号
  Private myXbisExitFlag As Boolean
  
  Private myXlonDataPrcsOptn As Long
    'myXlonDataPrcsOptn = 1 : 複数の元ファイル→1個のファイルへコピペ
    'myXlonDataPrcsOptn = 2 : 1個の元ファイル →複数のファイルへコピペ
    'myXlonDataPrcsOptn = 3 : 複数の元ファイル → 複数のファイルへコピペ (1対1)
  
'//モジュール内変数_データ
  Private myXobjSrchSheet As Object, myXstrSrchVal As String
  Private myXbisCmntSrch As Boolean, myXbisInStrOptn As Boolean

'iniP_モジュール内変数を初期化する
Private Sub initializeModuleVariables()
    myXbisExitFlag = False
    
    myXlonDataPrcsOptn = Empty
    
    Set myXobjSrchSheet = Nothing: myXstrSrchVal = Empty
    myXbisCmntSrch = False: myXbisInStrOptn = False
End Sub

'-----------------------------------------------------------------------------------------------

'PublicP_
Public Sub exeProc()
    
'//処理実行
    Call ctrProc
    
'//処理結果表示
    Select Case myXbisCmpltFlag
        Case True: MsgBox "実行完了"
        Case Else: MsgBox "異常あり", vbExclamation
    End Select
    
End Sub

'CtrlP_
Private Sub ctrProc()
    Call initializeOutputVariables
    Call initializeModuleVariables
'    Debug.Print "PassFlag: " & meMstrMdlName & "1"     'PassFlag
    
'//C:制御用変数を設定
    Call setControlVariables
    
'//S:シート上に記載されたデータから処理方法を取得
    Call snsProc
    If myXbisExitFlag = True Then GoTo ExitPath
'    Debug.Print "PassFlag: " & meMstrMdlName & "2"     'PassFlag
    
'//P:複数ファイルに対して連続処理
    Call prsProc
    If myXbisExitFlag = True Then GoTo ExitPath
'    Debug.Print "PassFlag: " & meMstrMdlName & "4"     'PassFlag
    
    myXbisCmpltFlag = True
ExitPath:
    Call initializeModuleVariables
End Sub

'iniP_出力変数を初期化する
Private Sub initializeOutputVariables()
    myXbisCmpltFlag = False
End Sub

'SetP_制御用変数を設定する
Private Sub setControlVariables()
    
    Set myXobjSrchSheet = ActiveSheet
    
    myXstrSrchVal = "処理方法の選択 ："
    
End Sub

'SnsP_シート上に記載されたデータから処理方法を取得
Private Sub snsProc()
    myXbisExitFlag = False
    
  Dim myXobjTrgtCell As Object
    Call instCTrgtCell(myXobjTrgtCell, myXobjSrchSheet, myXstrSrchVal, 0, 1, False, False)
    If myXobjTrgtCell Is Nothing Then GoTo ExitPath
    
    myXlonDataPrcsOptn = CLng(myXobjTrgtCell.Value)
    If myXlonDataPrcsOptn < 1 Or myXlonDataPrcsOptn > 3 Then GoTo ExitPath
    
    Set myXobjTrgtCell = Nothing
    Exit Sub
ExitPath:
    myXbisExitFlag = True
End Sub

'PrcsP_複数ファイルに対して連続処理
Private Sub prsProc()
    myXbisExitFlag = False
    
  Dim myXbisCompFlag As Boolean
    Select Case myXlonDataPrcsOptn
    '//複数の元ファイル→1個のファイルへコピペ
        Case 1
            Call bMsub1MultiToOne.callProc(myXbisCompFlag)
        
    '//1個の元ファイル→複数のファイルへコピペ
        Case 2
            Call bMsub2OneToMulti.callProc(myXbisCompFlag)
        
    '//複数の元ファイル→複数のファイルへコピペ(1対1)
        Case 3
            Call bMsub3MultiToMulti.callProc(myXbisCompFlag)
        
    End Select
    If myXbisCompFlag = False Then GoTo ExitPath
    
    Exit Sub
ExitPath:
    myXbisExitFlag = True
End Sub

'===============================================================================================

'◆ClassProc名_シート上のデータから文字列を検索してセルを取得する
Private Sub instCTrgtCell(myXobjTrgtCell As Object, _
                ByVal myXobjSrchSheet As Object, ByVal myXstrSrchVal As String, _
                ByVal myXlonRowOfst As Long, ByVal myXlonColOfst As Long, _
                Optional ByVal coXbisCmntSrch As Boolean = False, _
                Optional ByVal coXbisInStrOptn As Boolean = False)
    'coXbisCmntSrch = False : シート上の文字列を検索する
    'coXbisCmntSrch = True  : シート上のコメント内容を検索する
    'coXbisInStrOptn = False : 指定文字列と一致する条件で検索する
    'coXbisInStrOptn = True  : 指定文字列を含む条件で検索する
  Dim myXinsTrgtCell As CTrgtCell: Set myXinsTrgtCell = New CTrgtCell
    With myXinsTrgtCell
    '//クラス内変数への入力
        .letCmntSrch = coXbisCmntSrch
        Set .setSrchSheet = myXobjSrchSheet
        .letSrchVal = myXstrSrchVal
        .letRowOfst = myXlonRowOfst
        .letColOfst = myXlonColOfst
        .letInStrOptn = coXbisInStrOptn
    '//クラス内プロシージャの実行とクラス内変数からの出力
        Set myXobjTrgtCell = .fncobjTargetCell
    End With
    Set myXinsTrgtCell = Nothing
End Sub
